

\subsection{基本データ型}

\begin{tabular}[h]{p{3cm} p{3cm} p{7cm}} \hline
  \multicolumn{1}{c}{型名} & \multicolumn{1}{c}{型} & \multicolumn{1}{c}{意
    味} \\ \hline\hline
  B  & char & 符号付き8ビット整数 \\
  H  & short & \\
  W  & long  & \\
  UB & unsigned char & \\
  UH & unsigned short & \\
  UW & unsigned long & \\
  VW & long     & \\
  VH & short    & \\
  VB & char     & \\
  VP & void *   & \\
  FP & void (*)() & \\
  INT & int & \\
  UINT & unsigned int & \\
  BOOL & enum { FALSE=0, TRUE=1 } & \\
  FN & short & \\
  ID & int & \\
  BOOL\_ID & int & \\
  HNO & int & \\
  RNO & int & \\
  NODE & int & \\
  ATR  & unsigned int & \\
  ER   & int & \\
  PRI  & int & \\
  TMO  & int & \\
  TCODE & UH   & \\ \hline
\end{tabular}
